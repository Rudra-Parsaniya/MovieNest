SELECT 
    r.RecId,
    m.MovieTitle,
    m.MovieGenre,
    m.ReleaseYear,
    m.Rating,
    m.Description,
    m.Duration,
    m.ImgUrl
FROM 
    RecMovies r
INNER JOIN 
    Movies m
    ON r.MovieId = m.MovieId
ORDER BY 
    r.RecId;
